To return multiple elements, wrap them in `React.Fragment`


```javascript
import React from 'react'

export default function HomePageTiles() {

    return (
        <React.Fragment>
            <div>Element 1</div>
            <div>Element 2</div>
        </React.Fragment>
    )
}
```
