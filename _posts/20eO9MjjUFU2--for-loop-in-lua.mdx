       io.write(tostring(buffer_list[i]).."\n")
    end
    
    
As log:

local function log(message)
    local log_file = io.open("log.log", "a")
    io.output(log_file)
    io.write(message.."\n")
    buffer_list = vim.api.nvim_list_bufs()
    for i=1, #buffer_list do
       io.write(tostring(buffer_list[i]).."\n")
    end
    io.close(log_file)
end
