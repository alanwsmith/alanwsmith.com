SELECT songs.displayTitle, queuedSongs.primaryKey, SUM(transactions.numberOfBits) as theBits
FROM queuedSongs, songs, transactions
WHERE queuedSongs.streamerId = 125363791
AND songs.primaryKey = queuedSongs.songPrimaryKey
AND queuedSongs.primaryKey = transactions.queuedSongPrimaryKey
GROUP BY songs.displayTitle, queuedSongs.primaryKey
