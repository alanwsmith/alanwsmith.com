```lua
local function open_new_file()
    vim.api.nvim_command(':e path/to/file.txt')
end

local open_floating_window = function()
    local main_window_id = vim.api.nvim_get_current_win()
    local pop_buffer_1 = vim.api.nvim_create_buf(false, true)
    local pop_window_1 = vim.api.nvim_open_win(pop_buffer_1, true, {
            style="minimal", relative='editor',
            row=3, col=3, width=30, height=6, border='single'
        }
    )
    vim.api.nvim_set_current_win(main_window_id)
end
```
