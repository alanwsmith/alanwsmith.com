This is an example of how to make TwitchIO commands case insensitive. 



```python
#!/usr/bin/env python3

from twitchio.ext import commands

oauth_token = "YOUR_OAUTH_TOKEN"
client_id = "YOUR_APP_CLIENT_ID"
bot_account = 'BOT_ACCOUNT_NAME'
channel_name = 'CHAT_TO_WATCH_ACCOUNT_NAME'

class Bot(commands.Bot):

    def __init__(self):
        super().__init__(
            irc_token=oauth_token,
            client_id=client_id,
            nick=bot_account,
            prefix='!',
            initial_channels=[channel_name]
        )

    async def event_message(self, message):
        tokens = message.content.split(" ")
        tokens[0] = tokens[0].lower()
        message.content = " ".join(tokens)
        await self.handle_commands(message)

    @commands.command(name='ping')
    async def my_command(self, ctx):
        await ctx.send(f"Case insensitive `!ping` received that doesn't mess with other capitalization: {ctx.message.content}")

if __name__ == '__main__':
    bot = Bot()
    bot.run()
```
