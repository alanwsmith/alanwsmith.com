local console_terminal
local console_window

local function log()
    print("Trying to log")
    console_buffer = vim.api.nvim_create_buf(false, true)
    console_window = vim.api.nvim_open_win(console_buffer, true,
        {style = "minimal",relative='win', row=5, col=5, width=40, height=4}
    )
    console_terminal = vim.api.nvim_open_term(console_buffer, {})
    vim.api.nvim_chan_send(console_terminal, "asdf")
end

return {
   log = log
}
