```python
#!/usr/bin/env python3

import keyring

from twitchio.ext import commands

oauth_token = "YOUR_APP_OAUTH_TOKEN"
client_id = "YOUR_APP_CLIENT_ID"
bot_account = 'YOUR_BOT_ACCOUNT_NAME'
channel_name = 'NAME_OF_CHANNEL_TO_WATCH'

class Bot(commands.Bot):

    def __init__(self):
        super().__init__(
            irc_token=oauth_token,
            client_id=client_id,
            nick=bot_account,
            prefix='!',
            initial_channels=[channel_name]
        )

    async def event_command_error(self, ctx, error):
        print(f"Errored on: {ctx.message.content}")
        if isinstance(error, commands.errors.CommandNotFound):
            await ctx.send(f"Sorry, {ctx.message.author.name}, I don't know how to do that.")


if __name__ == '__main__'
    bot = Bot()
    bot.run()
```
