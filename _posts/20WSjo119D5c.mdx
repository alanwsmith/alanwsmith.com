This is another example with just two files that doesn't use the autoload




`~/.config/nvim/plugin/example_basic_call.vim`


    if exists('g:loaded_example_basic_call') | finish | endif
    
    let s:save_cpo = &cpo
    set cpo&vim
    
    command! ExampleBasicCall lua require'example_basic_call'.example_basic_call()
    
    let &cpo = s:save_cpo
    unlet s:save_cpo
    
    let g:loaded_example_basic_call = 1


### File: ~/.config/nvim/lua/example_basic_call.lua


    local function print_here()
        vim.api.nvim_command(':lua print("HERE")')
    end
    
    local function example_basic_call()
        print_here()
    end
    
    return {
      example_basic_call = example_basic_call,
    }
