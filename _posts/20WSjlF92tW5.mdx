title: How To Show The Current Git Branch In A ZSH Prompt
type: post
---

```command
function show_git_branch() {
    BRANCH_NAME=$(git branch --show-current 2> /dev/null)
    if [ $? -eq 0 ]
    then
        echo "[branch: $BRANCH_NAME]"
    fi
}


function precmd() {
    PROMPT="%~ $(show_git_branch) $ "
}
```
