Basic Example 

```python3
#!/usr/bin/env python3

import keyring
import mysql.connector

cnx = mysql.connector.connect(
    database=keyring.get_password('mysql-database', 'local_username'),
    host=keyring.get_password('mysql-host', 'local_username'),
    user=keyring.get_password('mysql-username', 'local_username'),
    password=keyring.get_password('mysql-password', 'local_username')
)

cursor = cnx.cursor()

query = (
    "SELECT * "
    "FROM playlists "
    "WHERE idString = %s"
)

idString = 'asdf1234'

cursor.execute(query, (idString, ))
    
for row in cursor:    
    print(row)

cnx.close()
```

And if you want more explicit error handling for connection issues, you can do this:


```python3
#!/usr/bin/env python3

import keyring
import mysql.connector


try:
    cnx = mysql.connector.connect(
        database=keyring.get_password('mysql-database', 'local_username'),
        host=keyring.get_password('mysql-host', 'local_username'),
        user=keyring.get_password('mysql-username', 'local_username'),
        password=keyring.get_password('mysql-password', 'local_username')
    )

except mysql.connector.Error as err:
    if err.errno == errorcode.ER_ACCESS_DENIED_ERROR:
        print("Something is wrong with your user name or password")
    elif err.errno == errorcode.ER_BAD_DB_ERROR:
        print("Database does not exist")
    else:
        print(err)
else:

    cursor = cnx.cursor()

    query = (
        "SELECT * "
        "FROM playlists "
        "WHERE idString = %s"
    )
    idString = 'asdf1234'
    cursor.execute(query, (idString, ))
    
    for (primaryKey, name, idString, json) in cursor:
        print(name)
    
    # Can also do
    #for row in cursor:
    #    print(row)

    print("Connected")
    cnx.close()

```
