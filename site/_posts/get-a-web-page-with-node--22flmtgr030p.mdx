```javascript
#!/usr/bin/env node

const https = require('https')

https
  .get('https://feeds.simplecast.com/xLr7FvDj', (res) => {
    res.on('data', (d) => {
        process.stdout.write(d)
    })
  })
  .on('error', (e) => {
    console.error(e)
  })
```

reference: https://nodejs.org/api/https.html#httpsgeturl-options-callback
