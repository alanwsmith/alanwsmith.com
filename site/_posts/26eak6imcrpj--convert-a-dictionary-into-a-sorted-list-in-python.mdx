genres = {
	"Junk": [1],
	"Chill": [2],
	"Dance": [3],
	"Jass": [4], 
	"A": [7],
	"B": [1],
	'a': [3],
	'c': [2], 
	"Adf": [],
	"Aeee": [],
	"aeeed": [],
	"Aerer": []
}

items = [{"id": f, "data": genres[f]} for f in sorted(genres.keys(), key=lambda x: x.lower())]

print(items)
